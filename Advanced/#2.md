typedef unsigned __int64 QWORD;

LRESULT CALLBACK WndProc(HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
	if (uMsg != WM_CREATE) {
		if (uMsg != WM_DESTROY) {
			return DefWindowProcA(hWnd, uMsg, wParam, lParam);
		}
		PostQuitMessage(0);
	}
	return 0;
}
